\subsection{Configurations}

\subsubsection{Hyperref}
External links are configured to be \colorname{RubineRed}, internal links are \colorname{RoyalBlue} and citations are \colorname{ForestGreen}.

\subsubsection{Tasks}
Tasks are enumerated by lower-case alphabets:

\begin{tasks}(4)
    \task Hello
    \task World
    \task Foo
    \task Bar
\end{tasks}

\subsubsection{Xcolor}
Three custom colours are introduced:

\begin{multicols}{3}
    \begin{itemize}
        \item \colorname{plotRed}
        \item \colorname{plotBlue}
        \item \colorname{plotGreen}
    \end{itemize}
\end{multicols}

\subsubsection{Tikz and Pgfplots}
\asdiasty{} automatically loads the following \texttt{tikz} libraries:

\begin{multicols}{3}
    \begin{itemize}
        \item \pkg{angles}{https://tikz.dev/library-angle}
        \item \pkg{arrows}{https://tikz.dev/library-arrows}
        \item \pkg{calc}{https://tikz.dev/library-calc}
        \item \pkg{decorations.markings}{https://tikz.dev/library-decorations\#pgf.decorations.markings}
        \item \pkg{intersections}{https://tikz.dev/tikz-coordinates\#pgf.intersections}

        \item \pkg{shapes}{https://tikz.dev/library-shapes}
